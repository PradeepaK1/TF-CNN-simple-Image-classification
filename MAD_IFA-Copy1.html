import tensorflow as tflw
from tensorflow import keras
import numpy as nump
import matplotlib.pyplot as matpltlib
import glob, os
import re
import PIL
from PIL import Image
def conv(loc, size):
        img = Image.open(loc).convert('L') 
        WIDTH, HEIGHT = img.size
        if WIDTH != HEIGHT:
                m_min_d = min(WIDTH, HEIGHT)
                img = img.crop((0, 0, m_min_d, m_min_d))
        # Scale the image to the requested maxsize by Anti-alias sampling.
        img.thumbnail(size, PIL.Image.ANTIALIAS)
        return nump.asarray(img)
def data(loc, size):
        images = []
        labels = []
        os.chdir(loc)
        for file in glob.glob("*.jpg"):
                img = conv(file, size)
                if re.match('bd.*', file):
                        images.append(img)
                        labels.append(0)
                elif re.match('nbd.*', file):
                        images.append(img)
                        labels.append(1)
        return (nump.asarray(images), nump.asarray(labels))
size = 100, 100
categ = ['bd', 'nbd']
(trainimg, trainlab) = data('C:/Users/Admin/Desktop/Dataset.jene/train', size)
(testimg, testlab) = data('C:/Users/Admin/Desktop/Dataset.jene/test', size)
trainimg = trainimg / 255.0
testimg = testimg / 255.0
model = keras.Sequential([
    keras.layers.Flatten(input_shape=(100, 100)),
        keras.layers.Dense(245, activation=tflw.nn.sigmoid),
        keras.layers.Dense(50, activation=tflw.nn.sigmoid),
    keras.layers.Dense(2, activation=tflw.nn.softmax),
    keras.layers.Flatten()
])
sgd = keras.optimizers.SGD(lr=0.01, decay=1e-5, momentum=0.7, nesterov=True)

model.compile(optimizer=sgd,
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])
score = model.predict(testimg)
model.fit(trainimg, trainlab, epochs=4)
testLoss, testAcc = model.evaluate(testimg, testlab)
print('Accuracy Score:', testAcc)
score = model.predict(testimg)
def display_images(images, labels):
        matpltlib.figure(figsize=(10,10))
        grid_size = min(25, len(images))
        for i in range(grid_size):
                matpltlib.subplot(5, 5, i+1)
                matpltlib.xticks([])
                matpltlib.yticks([])
                matpltlib.grid(False)
                matpltlib.imshow(images[i], cmap=matpltlib.cm.binary)
                matpltlib.xlabel(categ[labels[i]])
display_images(testimg, nump.argmax(score, axis = 1))
matpltlib.show()
